157  ::etl::expected<uint16_t, HRESULT>>::type

159  {
160  HRESULT ret = HRESULT::NotRegistered;
161  if (Base::_proxy->isInitialized())

174  }

308  ::etl::expected<uint16_t, HRESULT>
309  set(SetterArgumentType const& payload, SetterCallback const& cbk)
310  {
311  ::etl::expected<uint16_t, HRESULT> result
312  = ::etl::unexpected<HRESULT>(::etl::in_place_t{}, HRESULT::NotRegistered);
313 
314  if (Base::_proxy->isInitialized())
315  {
316  ::etl::optional<uint16_t> reqId = _setDispatcher.obtainRequestId(cbk);
317  if (reqId.has_value())
318  {
319  Message msg = Base::_proxy->generateMessageHeader(
320  SetterTraits::METHOD_MEMBER_ID, reqId.value());
321  HRESULT res = MessagePayloadBuilder::getInstance().allocate(payload, msg);
322  if (res == HRESULT::Ok)
323  {
324  res = Base::_proxy->sendMessage(msg);
325  if (res == HRESULT::Ok)
326  {
327  result = reqId.value();
328  }
329  else
330  {
331  MessagePayloadBuilder::deallocate(msg);
332  logger::logMessageSendingFailure(
333  logger::LogLevel::Error, logger::Error::SendMessage, res, msg);
334  result = ::etl::unexpected<HRESULT>(::etl::in_place_t{}, res);
335  }
336  }
337  else

339  result = ::etl::unexpected<HRESULT>(::etl::in_place_t{}, res);
340  }
341 
342  if (!result.has_value())

344  static_cast<void>(_setDispatcher.cancelRequest(reqId.value()));
345  }
346  }
347  else
348  {
349  result

351  }
352  }
353
