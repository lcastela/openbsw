55  void dataReceived(
56  ::udp::AbstractDatagramSocket& /* socket */,
57  ::ip::IPAddress sourceAddress,
58  uint16_t /* sourcePort */,
59  ::ip::IPAddress /* destinationAddress */,
60  uint16_t length) override
61  {
62  if (!_remoteIpAddresses.empty())
63  {
64  auto const sourceIpAddress = ::ip::packed(sourceAddress);
65  auto remoteIpAddresses = _remoteIpAddresses;
66  bool isValidAddress = false;
67  while (!remoteIpAddresses.empty())
68  {
69  auto const remoteIpAddress
70  = remoteIpAddresses.take<uint8_t const>(sourceIpAddress.size());
71  if (::etl::mem_compare(
72  sourceIpAddress.begin(), sourceIpAddress.end(), remoteIpAddress.begin())
73  == 0)
74  {
75  isValidAddress = true;
76  break;
77  }
78  }
79 
80  if (!isValidAddress)
81  {
82  (void)_socket.read(nullptr, length);
83 
84  _errorHandler(
85  ::routing::ErrorHandler::StatusCode::INVALID_REMOTE_IP_ADDRESS, sourceAddress);
86 
87  ++_invalidIpAddressPdus;
88 
89  return;
90  }
91  }
92 
93  auto const frame = _output.allocate(length);
94  if (frame.size() != length)
95  {
96  (void)_socket.read(nullptr, length);
97 
98  if (!_lastAllocationFailed)
99  {
100  _errorHandler(::routing::ErrorHandler::StatusCode::MEM_ALLOCATION_FAILURE);
101  _lastAllocationFailed = true;
102  }
103 
104  ++_failedMemAllocPdus;
105 
106  return;
107  }
108  _lastAllocationFailed = false;
109 
110  (void)_socket.read(frame.data(), length);
111 
112  _output.commit();
113  }
