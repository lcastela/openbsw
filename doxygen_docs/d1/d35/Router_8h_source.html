83 template<uint8_t MAX_NUM_CHANNELS>
84 void Router<MAX_NUM_CHANNELS>::init(
85  ::routing::PduRoutingTable const& table,
86  ::etl::span<::io::IReader*> readers,
87  ::etl::span<::io::IWriter*> writers)
88 {
89  if (_initialized)
90  {
91  ::util::logger::Logger::warn(::util::logger::ROUTING, "Router already initialized");
92  return;
93  }
94 
95  size_t const numberOfChannels = ::etl::min(
96  static_cast<size_t>(MAX_NUM_CHANNELS), ::etl::min(readers.size(), writers.size()));
97  readers = readers.first(numberOfChannels);
98  writers = writers.first(numberOfChannels);
99 
100  if ((::etl::find(readers.begin(), readers.end(), nullptr) != readers.end())
101  || (::etl::find(writers.begin(), writers.end(), nullptr) != writers.end()))
102  {
103  ::util::logger::Logger::warn(
104  ::util::logger::ROUTING, "Invalid reader(s) or writer(s) at initialization");
105  }
106 
107  if ((table.destinations.empty()) || (table.destinationOffsets.empty())
108  || (table.outputMessageIds.empty()))
109  {
110  ::util::logger::Logger::error(
111  ::util::logger::ROUTING, "Failed to initialize router: empty routing configuration");
112  return;
113  }
114 
115  if (table.destinations.size() != table.outputMessageIds.size())
116  {
117  ::util::logger::Logger::error(
118  ::util::logger::ROUTING,
119  "Failed to initialize router: destinations and output message IDs have different "
120  "sizes");
121  return;
122  }
123 
124  _table = table;
125  _readers = readers;
126  _writers = writers;
127 
128  _initialized = true;
129 }

151 template<uint8_t MAX_NUM_CHANNELS>
152 bool Router<MAX_NUM_CHANNELS>::run()
153 {
154  if (!_initialized)
155  {
156  ::util::logger::Logger::error(::util::logger::ROUTING, "Router uninitialized");
157  return false;
158  }
159 
160  for (size_t i = 0; i < _readers.size(); ++i)
161  {
162  auto const reader = _readers[_currentChannelId];
163  bool const pduRouted
164  = reader != nullptr ? route(_currentChannelId, _table, *reader, _writers) : false;
165  _currentChannelId = static_cast<size_t>(_currentChannelId) + 1U < _readers.size()
166  ? _currentChannelId + 1U
167  : 0U;
168  if (pduRouted)
169  {
170  return true;
171  }
172  }
173 
174  return false;
175 }
