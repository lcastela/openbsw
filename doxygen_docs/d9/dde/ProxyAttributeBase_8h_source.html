55  if (_proxy->isInitialized())

57  ::etl::optional<uint16_t> const reqId = _getDispatcher.obtainRequestId(attrCb);
58  if (reqId.has_value())
59  {
60  Message msg = _proxy->generateMessageHeader(
61  DispatcherTraits::METHOD_MEMBER_ID, reqId.value());
62  HRESULT const sendResult = _proxy->sendMessage(msg);
63  if (sendResult == HRESULT::Ok)
64  {
65  ret = reqId.value();
66  }
67  else
68  {
69  static_cast<void>(_getDispatcher.cancelRequest(reqId.value()));
70  ret = ::etl::unexpected<HRESULT>{sendResult};

72  logger::LogLevel::Error, logger::Error::SendMessage, ret.error(), msg);
73  }
74  }

78  }
