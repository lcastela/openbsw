68  uint32_t size() const
69  {
70  // Snapshot both cursors once to avoid a race between the comparison
71  // and the subtraction in the ternary expression below.
72  // acquire on _sent synchronises with the release in publishSlot() so that
73  // the payload written before publishSlot() is visible before peek().
74  uint32_t const txPos = _sent.load(::etl::memory_order_acquire);
75  uint32_t const rxPos = _received.load(::etl::memory_order_relaxed);
76  return (txPos >= rxPos) ? (txPos - rxPos) : (txPos + (2U * _maxSize)) - rxPos;
77  }
78 
80  bool isFull() const
81  {
82  uint32_t const rxPos = _received.load(::etl::memory_order_relaxed);
83  return _sent.load(::etl::memory_order_relaxed) == ((rxPos + _maxSize) % (2U * _maxSize));
84  }
85 
87  bool isEmpty() const
88  {
89  // acquire on _sent: see size() comment.
90  return _sent.load(::etl::memory_order_acquire)
91  == _received.load(::etl::memory_order_relaxed);
92  }

101  void takeSnapshot()
102  {
103  uint32_t const currentSize = size();
104  if (0U != currentSize)
105  {
106  _stats.loadSnapshot += static_cast<uint16_t>(currentSize);
107  ++_stats.processingCounter;
108  }
109  _stats.realLoadSnapshot += static_cast<uint16_t>(currentSize);
110  ++_stats.realProcessingCounter;
111  if (_stats.previousSnapshot == 0U)
112  {
113  _stats.maxFillRate = static_cast<uint8_t>(currentSize);
114  _stats.previousSnapshot = static_cast<uint8_t>(currentSize);
115  }
116  else
117  {
118  if (currentSize > _stats.previousSnapshot)
119  {
120  auto const diff = (currentSize - _stats.previousSnapshot);
121  if (diff > _stats.maxFillRate)
122  {
123  _stats.maxFillRate = static_cast<uint8_t>(diff);
124  }
125  }
126  _stats.previousSnapshot = static_cast<uint8_t>(currentSize);
127  }
128  }

166  ::etl::optional<WriteToken> reserveSlot()
167  {
168  ::etl::optional<WriteToken> token{};
169  if (isFull())
170  {
171  ++_stats.lostMessages;
172  }
173  else
174  {
175  uint32_t const sentVal = _sent.load(::etl::memory_order_relaxed);
176  uint32_t const nextSent = (sentVal + 1U) % (2U * _maxSize);
177  if (0U == _received.load(::etl::memory_order_relaxed))
178  {
179  ++_stats.startupLoad;
180  }
181  token.emplace(WriteToken{sentVal % _maxSize, nextSent});
182  }
183  return token;
184  }
185 
194  void publishSlot(WriteToken const& token)
195  {
196  _sent.store(token.nextSent, ::etl::memory_order_release);
197  uint32_t const currentSize = size();
198  if (currentSize > _stats.maxLoad)
199  {
200  _stats.maxLoad = static_cast<uint8_t>(currentSize);
201  }
202  }
