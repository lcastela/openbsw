171  template<typename T>
172  [[nodiscard]] HRESULT
173  emplaceExternalPayload(T const& obj, Message& msg, uint8_t const numberOfReferences)
174  {
175  static_assert(::etl::is_copy_constructible<T>::value, "T must have a copy constructor!");
176 
177  HRESULT ret = HRESULT::CannotAllocatePayload;
178 
179  uint16_t const sid = msg.getHeader().serviceId;
180  uint8_t* const buffer
181  = msg.isEvent() ? memory::getAllocSharedFunction(sid)(sizeof(T), numberOfReferences)
182  : memory::getAllocFunction(sid)(sizeof(T));
183 
184  if (buffer != nullptr)
185  {
186  ::etl::construct_object_at(buffer, obj);
187  auto const offset = static_cast<ptrdiff_t>(
188  ::etl::distance(memory::getRegionStartFunction(sid)(), buffer));
189  msg.setExternalPayload(offset, static_cast<uint32_t>(sizeof(T)));
190  ret = HRESULT::Ok;
191  }
192  else
193  {
194  logger::logAllocationFailure(
195  logger::LogLevel::Error,
196  logger::Error::Allocation,
197  ret,
198  msg,
199  static_cast<uint32_t>(sizeof(T)));
200  }
201 
202  return ret;
203  }
