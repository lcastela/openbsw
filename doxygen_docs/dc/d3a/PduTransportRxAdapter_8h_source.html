88 template<size_t MAX_ELEMENT_SIZE>
89 ErrorHandler::StatusCode
90 PduTransportRxAdapter<MAX_ELEMENT_SIZE>::processMessage(::etl::span<uint8_t const> frame) const
91 {
92  if (frame.size() < MESSAGE_HEADER_SIZE)
93  {
94  _currentMessageSize = frame.size();
95 
96  ++_invalidHeaderSizePdus;
97 
98  return ErrorHandler::StatusCode::INVALID_MESSAGE_HEADER_SIZE;
99  }
100 
101  auto header = frame.first(MESSAGE_HEADER_SIZE);
102 
103  (void)header.take<::etl::be_uint32_t const>(); // message ID
104  auto const parsedPayloadLength = header.take<::etl::be_uint32_t const>();
105  auto const messageSize = MESSAGE_HEADER_SIZE + parsedPayloadLength;
106  if (messageSize > frame.size())
107  {
108  _currentMessageSize = frame.size();
109 
110  ++_invalidParsedPayloadLengthPdus;
111 
112  return ErrorHandler::StatusCode::INVALID_PARSED_LENGTH;
113  }
114  _currentMessageSize = messageSize;
115 
116  auto const message = frame.take<uint8_t const>(_currentMessageSize);
117  return lookup(message);
118 }

120 template<size_t MAX_ELEMENT_SIZE>
121 ::etl::span<uint8_t> PduTransportRxAdapter<MAX_ELEMENT_SIZE>::peek() const
122 {
123  // Return the last PDU extracted.
124  if (!_currentPdu.empty())
125  {
126  return _currentPdu;
127  }
128 
129  if (_currentFrame.empty())
130  {
131  _currentFrame = _inputReader.peek();
132  }
133 
134  while (!_currentFrame.empty())
135  {
136  bool const isNewMessage = _currentPduIndex == 0;
137  if (isNewMessage)
138  {
139  auto const statusCode = processMessage(_currentFrame);
140 
141  ++_pduCounter;
142  _byteCounter += static_cast<uint32_t>(_currentMessageSize);
143 
144  if (statusCode != ErrorHandler::StatusCode::OK)
145  {
146  _errorHandler(statusCode, _currentFrame.first(_currentMessageSize));
147 
148  ++_discardedPduCounter;
149  _discardedByteCounter += static_cast<uint32_t>(_currentMessageSize);
150 
151  _currentFrame.advance(_currentMessageSize);
152 
153  if (_currentFrame.empty())
154  {
155  _inputReader.release();
156  _currentFrame = _inputReader.peek();
157  }
158 
159  continue;
160  }
161  }
162 
163  bool noPduExtracted = true;
164  auto const numberOfPdus = _table.pduLengthsOffsets[_currentMessageIndex + 1]
165  - _table.pduLengthsOffsets[_currentMessageIndex];
166  auto const message = _currentFrame.first(_currentMessageSize);
167  while (noPduExtracted && (_currentPduIndex < numberOfPdus))
168  {
169  auto const channelSpecificPduId
170  = _table.pduLengthsOffsets[_currentMessageIndex] + _currentPduIndex;
171  auto pdu = _buffer;
172  if (extract(message, channelSpecificPduId, pdu))
173  {
174  _currentPdu = pdu;
175  noPduExtracted = false;
176  }
177 
178  ++_currentPduIndex;
179  }
180 
181  if (_currentPduIndex == numberOfPdus)
182  {
183  _currentPduIndex = 0;
184  _currentFrame.advance(_currentMessageSize);
185 
186  if (_currentFrame.empty())
187  {
188  _inputReader.release();
189  _currentFrame = _inputReader.peek();
190  }
191  }
192 
193  if (noPduExtracted)
194  {
195  if (isNewMessage)
196  {
197  // Although this message was expected, no PDU was extracted from it because all the
198  // expected PDUs were too big
199  _errorHandler(ErrorHandler::StatusCode::INVALID_PAYLOAD_LENGTH, message);
200 
201  ++_invalidPayloadLengthPdus;
202 
203  ++_discardedPduCounter;
204  _discardedByteCounter += static_cast<uint32_t>(_currentMessageSize);
205  }
206 
207  continue;
208  }
209 
210  return _currentPdu;
211  }
212  return {};
213 }

215 template<size_t MAX_ELEMENT_SIZE>
216 ErrorHandler::StatusCode
217 PduTransportRxAdapter<MAX_ELEMENT_SIZE>::lookup(::etl::span<uint8_t const> message) const
218 {
219  auto const messageId = message.take<::etl::be_uint32_t const>();
220  auto const pos = ::etl::find(_table.messageIds.begin(), _table.messageIds.end(), messageId);
221  if (pos == _table.messageIds.end())
222  {
223  ++_unknownMessagePdus;
224 
225  return ErrorHandler::StatusCode::UNKNOWN_MESSAGE_ID;
226  }
227 
228  uint32_t const parsedPayloadLength = message.take<::etl::be_uint32_t const>();
229 
230  auto const index = static_cast<size_t>(::etl::distance(_table.messageIds.begin(), pos));
231  uint32_t const messageLength = (!_table.messageLengths.empty())
232  ? static_cast<uint32_t>(_table.messageLengths[index])
233  : 0U;
234 
235  if ((messageLength != 0) && (parsedPayloadLength != messageLength))
236  {
237  ++_invalidMessageLengthPdus;
238 
239  return ErrorHandler::StatusCode::INVALID_MESSAGE_LENGTH;
240  }
241 
242  _currentMessageIndex = index;
243 
244  return ErrorHandler::StatusCode::OK;
245 }
