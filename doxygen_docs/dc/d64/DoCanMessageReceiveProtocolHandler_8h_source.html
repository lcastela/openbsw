283 template<typename FrameIndexType>
284 inline ReceiveResult DoCanMessageReceiveProtocolHandler<FrameIndexType>::allocated(
285  bool const success, uint8_t const maxRetryCount)
286 {
287  if (!_isAllocating)
288  {
289  return ReceiveResult(true).setMessage(
290  ReceiveMessage::ILLEGAL_STATE, static_cast<uint8_t>(_state));
291  }
292 
293  if (success)
294  {
295  _isAllocating = false;
296  _allocateRetryCount = 0U;
297  return setState((_frameCount == 1U) ? ReceiveState::PROCESSING : ReceiveState::SEND);
298  }
299 
300  if (_frameCount == 1U)
301  {
302  ++_allocateRetryCount;
303  if (_allocateRetryCount > 1U)
304  {
305  return setDone(ReceiveMessage::ALLOCATION_RETRY_COUNT_EXCEEDED);
306  }
307 
308  return setState(ReceiveState::WAIT, ReceiveTimeout::ALLOCATE);
309  }
310 
311  ++_allocateRetryCount;
312  if (_allocateRetryCount > maxRetryCount)
313  {
314  return setDone(ReceiveMessage::ALLOCATION_RETRY_COUNT_EXCEEDED);
315  }
316 
317  return setState(ReceiveState::SEND);
318 }

338 template<typename FrameIndexType>
339 inline ReceiveResult DoCanMessageReceiveProtocolHandler<FrameIndexType>::consecutiveFrameReceived(
340  uint8_t const frameIndex, uint8_t const maxBlockSize)
341 {
342  if ((_state != ReceiveState::WAIT) || (_timeout != ReceiveTimeout::RX))
343  {
344  return ReceiveResult(false);
345  }
346 
347  if ((_frameIndex & 0xFU) != frameIndex)
348  {
349  return setDone(ReceiveMessage::BAD_SEQUENCE_NUMBER, frameIndex);
350  }
351 
352  ++_frameIndex;
353  if (_frameIndex == _frameCount)
354  {
355  return setState(ReceiveState::PROCESSING);
356  }
357  if (maxBlockSize > 0U)
358  {
359  ++_blockFrameIndex;
360  if (_blockFrameIndex == maxBlockSize)
361  {
362  _blockFrameIndex = 0U;
363  return setState(ReceiveState::SEND);
364  }
365  }
366  // set state to wait for timeout as in the default case
367  return setState(ReceiveState::WAIT, ReceiveTimeout::RX);
368 }
